Prolapse API by atomkern

http://localhost/prolapse

Node.JS Example:

const fetch = require('node-fetch');
    fetch('http://localhost/prolapse', {
        headers: {
          'Content-Type': 'application/json'
        }
      }).catch((err) => {
        console.log(err.message)
      })
      .then(res => res.json())
      .then(json => {
        console.log(json.url)
    })
